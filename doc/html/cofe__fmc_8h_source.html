140 template <int DIM> inline
141 cofe::TensorRank2<DIM> inverseOf(const cofe::TensorRank2<DIM> & T)
142 {
143  cofe::TensorRank2<DIM> r;
144  double det,invdet;
145  det = T.determinant();
146  if(std::abs(det) <= std::numeric_limits<double>::min()) cofe::CofeUtils::error("Near singular matrix");
147  invdet= 1.0/det;
148 
149  r(0,0)= T(1,1)*T(2,2) - T(1,2)*T(2,1);
150  r(1,0)= T(1,2)*T(2,0) - T(1,0)*T(2,2);
151  r(2,0)= T(1,0)*T(2,1) - T(1,1)*T(2,0);
152 
153  r(0,1)= T(0,2)*T(2,1) - T(0,1)*T(2,2);
154  r(1,1)= T(0,0)*T(2,2) - T(0,2)*T(2,0);
155  r(2,1)= T(0,1)*T(2,0) - T(0,0)*T(2,1);
156 
157  r(0,2)= T(0,1)*T(1,2) - T(0,2)*T(1,1);
158  r(1,2)= T(0,2)*T(1,0) - T(0,0)*T(1,2);
159  r(2,2)= T(0,0)*T(1,1) - T(0,1)*T(1,0);
160 
161  r *= invdet;
162  return r;
163 }

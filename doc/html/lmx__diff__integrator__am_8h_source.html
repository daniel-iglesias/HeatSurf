136  void IntegratorAM<T>::advance( )
137  {
138  int i, j;
139  if (q->getTimeSize() < order/*+1*/){ //orden de integrador-1 > steps
140  for ( i = 0; i < q->getDiffOrder(); ++i ){
141  // f{n} = (q{n} - q{n-1}) / ( h*b_0 ) + ...
142  q->setConf( i+1,( q->getConf(i,0) - q->getConf(i,1) )
143  * ( 1. / (q->getLastStepSize() * b[q->getTimeSize()-2][0])) );
144  for ( j=1; j < q->getTimeSize()-1; ++j){
145  // ... - b_i/b_0 * f{n-i}
146  q->setConf( i+1, q->getConf(i+1,0) - ( b[q->getTimeSize()-2][j]* q->getConf(i+1,j) )
147  * ( 1. / ( b[q->getTimeSize()-2][0])) );
148  }
149  }
150  }
151  else{//se aplica el orden real de integrador
152  for ( i = 0; i < q->getDiffOrder(); ++i ){
153  // f{n} = (q{n} - q{n-1}) / ( h*b_0 ) + ...
154  q->setConf( i+1,( q->getConf(i,0) - q->getConf(i,1) )
155  * ( 1. / (q->getLastStepSize() * b[order-1][0])) );
156  for ( j=1; j < order; ++j){
157  // ... - b_i/b_0 * f{n-i}
158  q->setConf( i+1, q->getConf(i+1,0) - ( b[order-1][j]* q->getConf(i+1,j) )
159  * ( 1. / ( b[order-1][0])) );

107  void mat_mat_mult( const Data_mat<T>* A,

109  Data_mat<T>* C)
110 {
111  // Emmit an error if C is A or B...

115  << endl << " This cannot be done." << endl;
116  LMX_THROW(failure_error, message.str() );
117  }
118  size_type i, j, k;
119  C->resize( A->getRows(), B->getCols() );
120  for (i=0; i< C->getRows(); ++i){
121  for (j=0; j< C->getCols(); ++j){
122  C->writeElement( T(0), i, j );
123  }
124  }
125  // This can have a good optimization...
126  for (i=0; i < C->getRows(); ++i){
127  for (j=0; j < C->getCols(); ++j){
128  C->writeElement( 0, i, j );
129  for (k=0; k < A->getCols(); ++k){
130  C->writeElement( C->readElement(i,j) + A->readElement(i,k) * B->readElement(k,j) , i, j );
131  }

297  if ( value != T(0) ){

302  }

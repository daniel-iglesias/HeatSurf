228 template<typename Sys, class T>
229 bool NLSolver<Sys, T>::convergence()
234 {
235  if (deltaInResidue == 0) { // use norm2 criteria
236  if (this->res_vector.norm2() < epsilon) { return 1; }
237  else { return 0; }
238  }
239  else { // use energetic criteria
240  if (iteration == 0) {
241  energy_0 = fabs(res_vector * q); // store first residual energy
242 // cout << "iteration, energy = "<< iteration << " "<< energy_0 << endl;
243  if (energy_0 < 1E-50) {
244  return 1; // too small energy, rare
245  } else { return 0; } // common return value
246  }
247  else { // rest of iterations
248  energy_i = fabs(res_vector * q);
249  energy_i /= energy_0; // dimensionless residual energy rate
250  if (energy_i < epsilon) {
251  return 1; // convergence!!
252  } else { return 0; } // not converged
253  }
254  }

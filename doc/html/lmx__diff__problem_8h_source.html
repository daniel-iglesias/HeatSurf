163 template<typename Sys, typename T>
164 void DiffProblem<Sys, T>::setIntegrator(int type, int opt1, int opt2)
165 {
166  switch (type) {
167  case 0 : // integrator == 0 -> Adams-Bashford
168  theIntegrator = new IntegratorAB<T>(opt1);
169  break;
170 
171  case 1 : // integrator == 1 -> Adams-Moulton
172  theIntegrator = new IntegratorAM<T>(opt1);
173  break;
174 
175  case 2 : // integrator == 2 -> BDF
176  theIntegrator = new IntegratorBDF<T>(opt1);
177  break;
178 
179  case 3 : // integrator == 2 -> BDF
180  theIntegrator = new IntegratorCentralDifference<T>();
181  break;
182 
183  }
184 
185 }

299 template<typename Sys, typename T>
300 void DiffProblem<Sys, T>::writeStepFiles()
301 {
302  std::map<int, std::ofstream *>::iterator it;
303  for (it = fileOutMap.begin(); it != fileOutMap.end(); ++it) {
304  it->second->setf(std::ios::scientific, std::ios::floatfield);
305  it->second->precision(6);
306  *(it->second) << theConfiguration->getTime() << "\t";
307  for (unsigned int i = 0; i < theConfiguration->getConf(it->first, 0).size(); ++i) {
308  *(it->second) << theConfiguration->getConf(it->first, 0).readElement(i) << "\t";
309  }
310  *(it->second) << endl;
311  }

151 template <typename T>
157  void LU<T>::lu()
158 {
159  int i;
160  int j;
161  int k;
162  T mult;
163  T aux1, aux2;
164  int numR = 0;
165  int numRaux;
166  Vector<T> vAux;
167  vAux.resize(dim);
168 
169  if (pivotFlag!=0)
170  {
171  for (j = 0; j<dim; j++) p(j) = j; // Initialize permutation vector
172  }
173 
174  for (k = 0; k<dim-1; k++)
175  {
176  if (pivotFlag!=0) // Pivoting
177  {
178 // aux1 = abs(mat.readElement(k,k));
179  aux1 = mat.readElement(k,k);
180  if(aux1 < T(0)) aux1 = -aux1; // templatized abs();
181  numR = k;
182  for (i = k+1; i<dim; i++)
183  {
184 // aux2 = abs(mat.readElement(i,k));
185  aux2 = mat.readElement(i,k);
186  if(aux2 < T(0)) aux2 = -aux2; // templatized abs();
187  if ( aux2 > aux1 )
188  {
189  numR = i;
190  aux1 = aux2;
191  }
192  }
193  // Interchange rows
194  for (j = 0; j<dim; j++) vAux(j) = mat.readElement(numR,j);
195  for (j = 0; j<dim; j++) mat.writeElement( mat.readElement(k,j),numR,j);
196  for (j = 0; j<dim; j++) mat.writeElement( vAux(j),k,j);
197  numRaux = p.readElement(numR);
198  p.writeElement(k,numR);
199  p.writeElement(numRaux,k);
200  }
201 
202  // This is not necessary if pivoting is employed
203  if ( mat.readElement(k,k) == 0 )
204  {
205  std::stringstream message;
206  message << "Null pivot diagonal term!!. Term position: " << k << ". Squared Matrix dimension: " << dim << "." << endl;
207  LMX_THROW(internal_error, message.str() );
208  }
209  //
210  for (i = k+1; i<dim; i++)
211  {
212  if( mat.readElement(i,k) != 0 )
213  {
214  mult = mat.readElement(i,k) / mat.readElement(k,k);
215  mat.writeElement(mult, i, k);
216  for (j = k+1; j<dim; j++)
217  {
218  mat(i,j) -= mult*mat.readElement(k,j);
219  }
220  }
221  }
222  }
223 }

234 {

237

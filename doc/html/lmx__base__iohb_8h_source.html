174  double substrtod(const char * p, size_type len, int Valflag)
175  {
176  char s[100];
177  len = std::min(len, sizeof s - 1);
178  strncpy(s, p, len);
179  s[len] = 0;
180  if (Valflag != 'F' && !strchr(s, 'E')) {
181  /* insert a char prefix for exp */
182  auto last = strlen(s);
183  for (auto j = last + 1; j >= 0; j--) {
184  s[j] = s[j - 1];
185  if (s[j] == '+' || s[j] == '-') {
186  s[j - 1] = (char)Valflag;
187  break;
188  }
189  }
190  }
191  return atof(s);

194  template<typename IND_TYPE>
195  int readHB_data(IND_TYPE colptr[], IND_TYPE rowind[],
196  double val[])
197  {
198  /************************************************************************/
199  /* This function opens and reads the specified file, interpreting its */
200  /* contents as a sparse matrix stored in the Harwell/Boeing standard */
201  /* format and creating compressed column storage scheme vectors to hold*/
202  /* the index and nonzero value information. */
203  /* */
204  /* ---------- */
205  /* **CAVEAT** */
206  /* ---------- */
207  /* Parsing real formats from Fortran is tricky, and this file reader */
208  /* does not claim to be foolproof. It has been tested for cases when */
209  /* the real values are printed consistently and evenly spaced on each */
210  /* line, with Fixed (F), and Exponential (E or D) formats. */
211  /* */
212  /* ** If the input file does not adhere to the H/B format, the ** */
213  /* ** results will be unpredictable. ** */
214  /* */
215  /************************************************************************/
216  int i, ind, col, offset, count;
217  int Ptrperline, Ptrwidth, Indperline, Indwidth;
218  int Valperline = 0, Valwidth = 0, Valprec, Nentries;
219  int Valflag = 0; /* Indicates 'E','D', or 'F' float format */
220  char line[BUFSIZ];
221 
222  /* Parse the array input formats from Line 3 of HB file */
223  ParseIfmt(Ptrfmt, &Ptrperline, &Ptrwidth);
224  ParseIfmt(Indfmt, &Indperline, &Indwidth);
225  if (Type[0] != 'P') { /* Skip if pattern only */
226  ParseRfmt(Valfmt, &Valperline, &Valwidth, &Valprec, &Valflag);
227  }
228 
229  /* Read column pointer array: */
230  offset = 0; /* if base 0 storage is declared (via macro def), */
231  /* then storage entries are offset by 1 */
232 
233  for (count = 0, i = 0; i < Ptrcrd; i++) {
234  getline(line);
235  for (col = 0, ind = 0; ind < Ptrperline; ind++) {
236  if (count > Ncol) break;
237  colptr[count] = substrtoi(line + col, (size_t)Ptrwidth) - offset;
238  count++;
239  col += Ptrwidth;
240  }
241  }
242 
243  /* Read row index array: */
244  for (count = 0, i = 0; i < Indcrd; i++) {
245  getline(line);
246  for (col = 0, ind = 0; ind < Indperline; ind++) {
247  if (count == Nnzero) break;
248  rowind[count] = substrtoi(line + col, (size_t)Indwidth) - offset;
249  count++;
250  col += Indwidth;
251  }
252  }
253 
254  /* Read array of values: */
255  if (Type[0] != 'P') { /* Skip if pattern only */
256  if (Type[0] == 'C') {
257  Nentries = 2 * Nnzero;
258  } else { Nentries = Nnzero; }
259 
260  count = 0;
261  for (i = 0; i < Valcrd; i++) {
262  getline(line);
263  if (Valflag == 'D') {
264  // const_cast Due to aCC excentricity
265  char * p;
266  while ((p = const_cast<char *>(strchr(line, 'D')))) *p = 'E';
267  }
268  for (col = 0, ind = 0; ind < Valperline; ind++) {
269  if (count == Nentries) break;
270  val[count] = substrtod(line + col, (size_t)Valwidth, Valflag);
271  count++;
272  col += Valwidth;
273  }
274  }
275  }
276  return 1;
277  }

281 inline void HarwellBoeing_IO::open(const char * filename)
282 {
283  int Totcrd, Neltvl, Nrhsix;
284  char line[BUFSIZ];
285  close();
286  f = fopen(filename, "r");
287  if (!f) {LMX_THROW(lmx::failure_error, "could not open " << filename); }
288  /* First line: */
289  sscanf(getline(line), "%72c%8s", Title, Key);
290  Key[8] = Title[72] = 0;
291  /* Second line: */
292  Totcrd = Ptrcrd = Indcrd = Valcrd = Rhscrd = 0;
293  sscanf(getline(line), "%d%d%d%d%d", &Totcrd, &Ptrcrd, &Indcrd, &Valcrd, &Rhscrd);
294 
295  /* Third line: */
296  Nrow = Ncol = Nnzero = Neltvl = 0;
297  if (sscanf(getline(line), "%3c%d%d%d%d", Type, &Nrow, &Ncol, &Nnzero, &Neltvl) < 1) {
298  IOHBTerminate("Invalid Type info, line 3 of Harwell-Boeing file.\n");
299  }
300  std::for_each(Type, Type + 3, (int (*)(int)) toupper);
301 // std::for_each(Type, Type+3, toupper);
302  /* Fourth line: */
303  if (sscanf(getline(line), "%16c%16c%20c%20c", Ptrfmt, Indfmt, Valfmt, Rhsfmt) < 3) {
304  IOHBTerminate("Invalid format info, line 4 of Harwell-Boeing file.\n");
305  }
306  Ptrfmt[16] = Indfmt[16] = Valfmt[20] = Rhsfmt[20] = 0;
307 
308  /* (Optional) Fifth line: */
309  if (Rhscrd != 0) {
310  Nrhs = Nrhsix = 0;
311  if (sscanf(getline(line), "%3c%d%d", Rhstype, &Nrhs, &Nrhsix) != 1) {
312  IOHBTerminate("Invalid RHS type information, line 5 of"
313  " Harwell-Boeing file.\n");
314  }
315  }

168 template<typename Sys, typename T>
169 void DiffProblemDouble<Sys, T>::setIntegrator1(int type, int opt1, int opt2)
170 {
171  switch (type) {
172  case 0 : // integrator == 0 -> Adams-Bashford
173  theIntegrator1 = new IntegratorAB<T>(opt1);
174  break;
175 
176  case 1 : // integrator == 1 -> Adams-Moulton
177  theIntegrator1 = new IntegratorAM<T>(opt1);
178  break;
179 
180  case 2 : // integrator == 2 -> BDF
181  theIntegrator1 = new IntegratorBDF<T>(opt1);
182  break;
183 
184  case 3 : // integrator == 2 -> BDF
185  theIntegrator1 = new IntegratorCentralDifference<T>();
186  break;
187  }
188 }

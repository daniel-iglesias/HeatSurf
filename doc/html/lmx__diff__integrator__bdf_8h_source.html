145  void IntegratorBDF<T>::advance( )
146  {
147  int i, j;
148  if (q->getTimeSize() < order+1){ //orden de integrador > steps
149  for ( i = 0; i < q->getDiffOrder(); ++i ){
150  q->setConf( i+1,( q->getConf(i,0) )
151  * ( 1. / (q->getLastStepSize() * b[q->getTimeSize()-2])) );
152  for ( j=0; j < q->getTimeSize()-1; ++j){
153  q->setConf( i+1, q->getConf(i+1,0) - ( a[q->getTimeSize()-2][j]* q->getConf(i,j+1) )
154  * ( 1. / (q->getLastStepSize() * b[q->getTimeSize()-2])) );
155  }
156  }
157  }
158  else{//se aplica el orden real de integrador
159  for ( i = 0; i < q->getDiffOrder(); ++i ){
160 // q->setConf( i+1,0).fillIdentity(0);
161  q->setConf( i+1,( q->getConf(i,0) )

163  for ( j=0; j < order; ++j){
164  q->setConf( i+1, q->getConf(i+1,0) - ( a[order-1][j]* q->getConf(i,j+1) )
